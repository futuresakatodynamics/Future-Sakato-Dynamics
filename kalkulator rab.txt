function calcKusenPintu(kode){
  const tipe=document.getElementById('tipe_'+kode)?.value||'tunggal';
  const l=parseFloat(document.getElementById('l'+kode)?.value)||0;
  const t=parseFloat(document.getElementById('t'+kode)?.value)||0;
  let kusen=tipe==='kupukupu'?(2*t+2*l):(2*t+l);
  setText('kusen'+kode,f3(kusen)+" m'");
  return{tipe,l,t,kusen};
}
function calcKusenJendela(kode){
  const l=parseFloat(document.getElementById('l'+kode)?.value)||0;
  const t=parseFloat(document.getElementById('t'+kode)?.value)||0;
  const kusen=2*(l+t);
  setText('kusen'+kode,f3(kusen)+" m'");
  return{l,t,kusen};
}
function updBiayaPintu(){
  let total=0;
  Object.keys(QP).forEach(k=>{
    const{tipe,l,t,kusen}=calcKusenPintu(k);
    const areaDaun=l*t*(tipe==='kupukupu'?2:1);
    total+=(kusen*HS.kusen_alum+areaDaun*HS.pintu_teakwood+HS.hardware_pintu)*(QP[k]||0);
  });
  setText('vBiayaP',rp(total));updMeter();
}
function updBiayaJendela(){
  let total=0;
  Object.keys(QJ).forEach(k=>{
    const{l,t,kusen}=calcKusenJendela(k);
    total+=(kusen*HS.kusen_alum+kusen*HS.rangka_alum+l*t*HS.kaca_6mm+HS.hardware_jendela)*(QJ[k]||0);
  });
  total+=(ST.BV1||1)*(2.56*HS.kusen_alum+0.34*HS.kaca_es_5mm+4.72*HS.clear_sealant);
  setText('vBiayaJ',rp(total));updMeter();
}
